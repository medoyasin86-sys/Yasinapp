# ملف: planner_2026_2027.py # أجندة رقمية عربية كاملة (يومي + أسبوعي + شهري) لعامي 2026-2027 from reportlab.pdfgen import canvas from reportlab.lib.pagesizes import A4 from reportlab.lib.colors import HexColor from reportlab.pdfbase.ttfonts import TTFont from reportlab.pdfbase import pdfmetrics import calendar import datetime # تسجيل خط عربي pdfmetrics.registerFont(TTFont('Amiri', 'Amiri-Regular.ttf')) def draw_header(c, text, y): c.setFont("Amiri", 24) c.setFillColor(HexColor("#c49b63")) c.drawCentredString(A4[0]/2, A4[1]-y, text) def daily_page(c, date): c.setFillColor(HexColor("#fdfaf6")) c.rect(0, 0, A4[0], A4[1], fill=True, stroke=False) # عنوان اليوم draw_header(c, f"اليوم: {date.strftime('%d %B %Y')}", 60) # أقسام الصفحة sections = [ ("أهداف اليوم", 150), ("خواطري", 300), ("مخططاتي", 450), ("اقتباس اليوم", 600) ] c.setFont("Amiri", 16) for title, y in sections: c.setFillColor(HexColor("#c49b63")) c.drawRightString(A4[0]-50, A4[1]-y, title) c.setStrokeColor(HexColor("#d9cbb6")) c.rect(50, A4[1]-y-30, A4[0]-100, 100, stroke=True, fill=False) c.showPage() def weekly_page(c, year, week_num): draw_header(c, f"الأسبوع {week_num} - {year}", 60) days = ["السبت","الأحد","الاثنين","الثلاثاء","الأربعاء","الخميس","الجمعة"] c.setFont("Amiri", 14) y = 150 for d in days: c.drawRightString(A4[0]-50, A4[1]-y, d) c.rect(50, A4[1]-y-30, A4[0]-100, 60, stroke=True, fill=False) y += 80 c.showPage() def monthly_page(c, year, month): draw_header(c, f"شهر {calendar.month_name[month]} {year}", 60) c.setFont("Amiri", 16) c.drawRightString(A4[0]-50, A4[1]-150, "أهداف الشهر") c.rect(50, A4[1]-180, A4[0]-100, 200, stroke=True, fill=False) c.showPage() def create_planner(filename): c = canvas.Canvas(filename, pagesize=A4) for year in [2026, 2027]: # صفحات شهرية for month in range(1, 13): monthly_page(c, year, month) # صفحات أسبوعية for week in range(1, 53): weekly_page(c, year, week) # صفحات يومية start_date = datetime.date(year, 1, 1) end_date = datetime.date(year, 12, 31) delta = datetime.timedelta(days=1) d = start_date while d <= end_date: daily_page(c, d) d += delta c.save() if __name__ == "__main__": create_planner("planner_2026_2027.pdf")
